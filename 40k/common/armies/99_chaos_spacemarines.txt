emperors_children = {
	damage = 3.0
	health = 8.0
	morale = 3.0
	collateral_damage = 1.00
	war_exhaustion = 0.0
	time = 360
	
	
	resources = {
		category = armies
		cost = {
			military_goods = 20
                        gene_seed = 10
		}

		upkeep = {
			energy = 5
		}
	}

	allow = {
		has_building = building_chaos_temple
		
		owner = {
			NOT = { 
				check_variable = {
					which = "emperors_children_count" 
					value = 6
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "emperors_children_count" 
				value = 1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "emperors_children_count" 
				value = -1
			}
		}
	}

	potential = {
		OR = {
			owner = { has_country_flag = the_emperor_children }
			AND = {
				owner = { has_country_flag = chaos_empire }
			}
		}
	}
}
