imperial_knight = {
	damage = 4
	health = 6
	morale = 4
	collateral_damage = 1
	war_exhaustion = 0
	time = 500
	resources = {
		category = armies
		cost = {
			alloys = 2000
			electronics = 100
			military_goods = 50
		}
		upkeep = {
			energy = 4.5
		}
	}
	prerequisites = {
		"tech_collegia_titanica"
	}
	allow = {
		has_building = building_machine_cult
		owner = {
			NOT = {
				check_variable = {
					which = "imperial_knight_count"
					value = 20
				}
			}
		}
	}
	on_queued = {
		owner = {
			change_variable = {
				which = "imperial_knight_count"
				value = 1
			}
		}
	}
	on_unqueued = {
		owner = {
			change_variable = {
				which = "imperial_knight_count"
				value = -1
			}
		}
	}
	potential = {
		planet = {
			OR = {
				is_planet_class = pc_mechanicus_forge
			}
		}
		owner = {
			has_country_flag = imperium
		}
	}
}
