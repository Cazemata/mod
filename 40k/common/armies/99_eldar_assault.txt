wraithguard = {
	damage = 11.0
	health = 12.0
	morale = 10.5
	collateral_damage = 1.00
	war_exhaustion = 0.0
	time = 1000
	
	
	resources = {
		category = armies
		cost = {
			alloys = 1000
			spirit_stones = 150
		}

		upkeep = {
			energy = 10
		}
	}
	
	prerequisites = { "tech_wraithbone_wraithguard" }
	
	allow = {
		has_building = building_spiritspire_dharquese
		
		owner = {
			NOT = { 
				check_variable = {
					which = "wraithguard_count" 
					value = 5
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "wraithguard_count" 
				value = 1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "wraithguard_count" 
				value = -1
			}
		}
	}
	
	potential = {
		owner = { has_country_flag = eldar_empire }
	}
}
