########### Agri World District Set ###########
district_40k_city = {
        icon = district_hive_housing
	base_buildtime = 480
	is_capped_by_modifier = no

	zone_slots = {
		slot_city_government
		slot_city_01
		slot_city_02
	}

	show_on_uncolonized = {
            uses_district_set = 40kagri_world
	}

	potential = {
            uses_district_set = 40kagri_world
	}

	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	conversion_ratio = 1.0
	convert_to = {
		district_industrial_city
                district_planetary_fortress
	}
    
	conversion_ratio = 0.5
	convert_to = {
		district_arcology_housing
		district_nexus
		district_hive
                district_planetary_fortress
	}

	resources = {
		category = planet_districts_cities
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 5000
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = -100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			job_clerk_add = 100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 100
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = {
			job_clerk_add = 100
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_clerk_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 100
			planet = {
				planet_devastation < 1
				years_passed > 10
				free_housing < 4
			}
		}
	}
}
