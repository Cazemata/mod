		uses_district_set = daemonk_world
	}

		owner = { has_country_flag = chaos_empire }

		cost = {
			minerals = 500
		} 
	}

	planet_modifier = {
		planet_housing_add = 500
		job_berserk_add = 500
	}

	triggered_desc = {
		text = job_berserk_effect_desc
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_country_flag = khorne }
		}
		modifier = {
			planet_army_build_speed_mult = 0.05
		}
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			free_housing < 10
		}
		modifier = {
			weight = 1
			free_jobs < 1
			owner = { used_naval_capacity_percent > 0.9	}
		}
		modifier = {
			weight = 100
			owner = { used_naval_capacity_percent > 0.8	}
		}
	}
	
	ai_resource_production = {
		gods_favor = 1
		trigger = {
			always = yes
		}
	}
