district_refinery_housing = {
        icon = district_hive_housing
	base_buildtime = 600
	is_capped_by_modifier = no

	zone_slots = {
		slot_city_government
		slot_city_01
		slot_city_02
	}

	show_on_uncolonized = {
		uses_district_set = forgee_planet
	}

	potential = {
        OR = {
            uses_district_set = forgee_planet
		}
        owner = {
			exists = owner
			is_regular_empire = yes
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_hive_housing
	}
    
	conversion_ratio = 0
	convert_to = {
        district_generator
        district_mining
        district_farming
        district_hive_housing
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}
	}

	planet_modifier = {
		planet_housing_add = 4000
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 300
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 300
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 300
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		modifier = {
			job_clerk_add = 500
			job_criminal_add = 100
                        planet_crime_add = 20
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = {
			job_clerk_add = 500
			job_criminal_add = 100
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			free_housing < 1000
		}
	}
}
